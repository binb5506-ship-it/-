## 一、串口（UART）模块

### 1.1 uart.c 正确写法

```c
#include "uart.h"
#include "stdio.h"      // ★ 必须包含，否则 printf 重定向不生效

void Uart1_Init(void)   // 9600bps @ 12.000MHz
{
    SCON  = 0x50;       // 8位数据，可变波特率
    AUXR |= 0x01;       // 串口1选Timer2为波特率发生器
    AUXR &= 0xFB;       // Timer2 时钟12T模式
    T2L   = 0xE6;       // 定时初始值
    T2H   = 0xFF;
    AUXR |= 0x10;       // 启动Timer2
    ES    = 1;          // 使能串口中断
    EA    = 1;
}

char putchar(char ch)   // ★ 不加 extern，重定向 printf 到串口
{
    SBUF = ch;
    while(TI == 0);
    TI = 0;
    return ch;
}
```

### 1.2 uart.h 正确写法

```c
#include <STC15F2K60S2.H>
void Uart1_Init(void);
// ★ 不要写 extern char putchar(char ch); 这行会干扰重定向
```

### 1.3 串口接收中断（ISR）正确写法

```c
idata unsigned char Uart_Rx_Index = 0;
pdata unsigned char Uart_Rx_Buf[10] = {0};
idata unsigned char Uart_Rx_Flag  = 0;
idata unsigned char Uart_Rx_Tick  = 0;

void Uart1_Isr(void) interrupt 4
{
    if(RI)
    {
        Uart_Rx_Flag = 1;
        Uart_Rx_Tick = 0;
        Uart_Rx_Buf[Uart_Rx_Index++] = SBUF;   // ★ 先存后++
        RI = 0;
        if(Uart_Rx_Index >= 10)                 // ★ >= 防止越界
        {
            Uart_Rx_Index = 0;
            memset(Uart_Rx_Buf, 0, 10);
        }
    }
}
```

### 1.4 串口超时处理（Uart_Proc）

```c
void Uart_Proc()
{
    if(Uart_Rx_Index == 0) return;      // 没有数据，直接返回
    if(Uart_Rx_Tick >= 10)              // 超过10ms没有新字节，认为一帧结束
    {
        Uart_Rx_Flag = 0;
        Uart_Rx_Tick = 0;

        // 解析命令
        if(strcmp(Uart_Rx_Buf, "Nice") == 0)
            printf("%bu", dat);
        else
            printf("error\r\n");

        memset(Uart_Rx_Buf, 0, Uart_Rx_Index);
        Uart_Rx_Index = 0;
    }
}
```

### 1.5 Timer1 ISR 里驱动超时计数

```c
void Timer1_Isr(void) interrupt 3
{
    uwTick++;
    // ... 数码管扫描 ...
    if(Uart_Rx_Flag)
        Uart_Rx_Tick++;     // 每1ms加一，用于超时判断
}
```

### 1.6 串口调试常见问题

| 现象 | 原因 | 解决 |
|------|------|------|
| 完全没有接收/发送 | uart.c 未加入 Keil 工程 | 右键 Source Group → Add Existing Files |
| printf 没有输出 | uart.c 缺 `#include "stdio.h"` | 加上这行 |
| 收到乱码 | 波特率不匹配 | STC-ISP 串口助手设为9600 |
| 发 "Nice" 无响应 | `Uart1_Init()` 未调用 | 在 main() 里加 |
| 发送多次后失效 | 缓冲区溢出（10字节只能存2.5个"Nice"）| 一次发一条，等回复再发 |
| strcmp 失败 | STC-ISP 勾选了"发送新行"，实际发的是"Nice\r\n" | 取消勾选 |

## 二、调度器框架

### 2.1 完整正确框架

```c
typedef struct
{
    void (*task_func)(void);
    unsigned long int rate_ms;
    unsigned long int last_ms;
} task_t;

idata task_t Scheduler_Task[] =         // ★ 必须注册任务，不能留空
{
    {Led_Proc,  1,   0},                // LED刷新，每1ms
    {Key_Proc,  10,  0},                // 按键扫描，每10ms
    {Seg_Proc,  20,  0},                // 数码管，每20ms
    {Uart_Proc, 10,  0},                // 串口处理，每10ms
    {Get_Time,  100, 0},                // 读时钟，每100ms（有DS1302时加）
};

idata unsigned char task_num;

void Scheduler_Init()
{
    task_num = sizeof(Scheduler_Task) / sizeof(task_t);
}

void Scheduler_Run()
{
    unsigned char i;
    for(i = 0; i < task_num; i++)       // ★ 必须有 for 循环
    {
        unsigned long int now_time = uwTick;
        if(now_time >= Scheduler_Task[i].last_ms + Scheduler_Task[i].rate_ms)
        {                               // ★ >= 不是 >
            Scheduler_Task[i].last_ms = now_time;
            Scheduler_Task[i].task_func();
        }
    }
}
```

### 2.2 main() 标准初始化顺序

```c
void main()
{
    System_Init();      // ① 端口初始化，防止上电乱显
    Set_Rtc(ucRtc);    // ② 有DS1302时设置初始时间
    Uart1_Init();       // ③ 串口初始化（有串口时）
    Timer0_Init();      // ④ 有频率计数时加
    Scheduler_Init();   // ⑤ 计算任务数量
    Timer1_Init();      // ⑥ 启动1ms定时器（最后启动，防止调度器未就绪就中断）
    while(1)
    {
        Scheduler_Run();
    }
}
```

## 三、按键检测

### 3.1 正确的边沿检测公式

```c
void Key_Proc()
{
    Key_Val = Key_Read();
    Key_Down = Key_Val & (Key_Val ^ Key_Old);   // 按下事件（上升沿）
    Key_Up   = ~Key_Val & (Key_Val ^ Key_Old);  // 抬起事件（下降沿）
    Key_Old  = Key_Val;
}
```

**注意**：`Key_Down` 和 `Key_Up` 的公式在不同参考代码中有两种写法：
- 写法A（多见）：`Key_Down = Key_Val & (Key_Val ^ Key_Old);`
- 写法B：`Key_Down = ~Key_Val & (Key_Val ^ Key_Old);`

两种写法高电平/低电平有效定义不同，**保持与自己模板一致即可，不要混用**。

### 3.2 按键值对应关系（CT107D 开发板）

| 按键 | Key_Down 值 |
|------|------------|
| S4   | 4          |
| S5   | 5          |
| S6   | 6          |
| S7   | 7          |

### 3.3 按次数控制 LED（计数变量法）

```c
idata unsigned char led_count = 0;

// Key_Proc 里
if(Key_Down == 4)
    led_count = (led_count + 1) % 9;   // 0→1→2→...→8→0

// Led_Proc 里
void Led_Proc()
{
    unsigned char i;
    for(i = 0; i < 8; i++)
        ucled[i] = (i < led_count) ? 1 : 0;
    led_disp(ucled);
}
```

**错误写法（不要这样写）**：
```c
// ❌ else if 里再判断 Key_Down==4 永远不会执行
if(Key_Down == 4) { ucled[0]=1; }
else if(Key_Down == 4 && ...) { ucled[1]=1; }   // 永远不到这里
```

## 四、本地/远程模式切换

```c
idata bit Mode = 0;     // 0=本地按键控制  1=远程串口控制

// Key_Proc 里
if(Key_Down == 7) Mode ^= 1;               // S7 切换模式
if(Key_Down == 4 && Mode == 0)             // 仅本地模式响应按键

// Uart_Proc 里

    else if(Uart_Rx_Buf[0] == '2') led_count = 2;
    else if(Uart_Rx_Buf[0] == '3') led_count = 1;
}
```

## 五、数码管显示

### 5.1 Seg_Buf 编码规则

| Seg_Buf 值 | 显示内容 |
|-----------|---------|
| 0~9       | 数字 0~9 |
| 10        | 熄灭（不显示）|
| 11        | 字母 C / U / F（取决于段码表）|
| 12        | 字母 P / F |
| 值 + ','  | 对应数字带小数点（例：`3 + ','` 显示 `3.`）|

### 5.2 时间显示写法

```c
// 显示 HH-MM-SS
Seg_Buf[0] = ucRtc[0] / 10;    // 时十位
Seg_Buf[1] = ucRtc[0] % 10;    // 时个位
Seg_Buf[2] = 0 + ',';          // 显示 '-'
Seg_Buf[3] = ucRtc[1] / 10;
Seg_Buf[4] = ucRtc[1] % 10;
Seg_Buf[5] = 0 + ',';
Seg_Buf[6] = ucRtc[2] / 10;
Seg_Buf[7] = ucRtc[2] % 10;
```

### 5.3 数值前导零消除写法

```c
// 显示数值，高位为0时熄灭
Seg_Buf[4] = (val / 1000 % 10 == 0) ? 10 : val / 1000 % 10;
Seg_Buf[5] = (val / 100  % 10 == 0 && Seg_Buf[4] == 10) ? 10 : val / 100 % 10;
Seg_Buf[6] = val / 10 % 10;
Seg_Buf[7] = val % 10;
```

## 七、DS1302 时钟模块

### 7.1 引脚定义

```c
sbit SCK = P1^7;    // 时钟线
sbit SDA = P2^3;    // 数据线
sbit RST = P1^3;    // 片选线
```

### 7.2 核心 API

```c
// 设置时间（只需上电调用一次）
// ucRtc[0]=时, ucRtc[1]=分, ucRtc[2]=秒，传十进制
void Set_Rtc(unsigned char *ucRtc);

// 读取当前时间（放调度器里定期调用）
// 读出的是十进制，直接用
void Read_Rtc(unsigned char *ucRtc);
```

### 7.3 使用三步法

```c
// ① 全局变量
pdata unsigned char ucRtc[3] = {11, 12, 13};   // 初始时间 11:12:13

// ② main() 里设置一次
Set_Rtc(ucRtc);

// ③ 调度器里定期读取
void Get_Time() { Read_Rtc(ucRtc); }
// Scheduler_Task 里加：{Get_Time, 100, 0}
```

### 7.4 BCD 码转换（驱动内已封装，了解即可）

```
写入：十进制 → BCD：val / 10 * 16 + val % 10
读出：BCD → 十进制：bcd / 16 * 10 + bcd % 16
```

## 八、今日发现的典型错误汇总

| 错误 | 位置 | 正确写法 |
|------|------|---------|
| `Scheduler_Init;` 漏括号 | main() | `Scheduler_Init();` |
| `Timer1_Init()` 未调用 | main() | 加上调用 |
| `System_Init()` 未调用 | main() | 第一行调用 |
| Scheduler_Run 无 for 循环 | Scheduler_Run() | 加 for 循环 |
| Scheduler_Task[] 为空 | 全局 | 注册所有任务 |
| `i` 未初始化直接使用 | Scheduler_Run() | for 循环自动赋值 |
| 用 `>` 而不是 `>=` | Scheduler_Run() | 改为 `>=` |
| Uart_Rx_Index 未自增 | UART ISR | 改为 `Buf[Index++] = SBUF` |
| 溢出判断用 `> 10` | UART ISR | 改为 `>= 10` |
| uart.c 缺 `#include "stdio.h"` | uart.c | 加上这行 |
| uart.h 多余的 putchar 声明 | uart.h | 删除 |
| uart.c 未加入 Keil 工程 | Keil 工程 | 右键添加文件 |
| putchar 定义加了 extern | uart.c | 去掉 extern |
| `ucled` 初始值非全零 | 全局变量 | 初始化为全0 |
| Seg_Buf 用 idata 而不是 pdata | 全局变量 | 改为 pdata |
| else if 里重复判断 Key_Down | Led_Proc | 用计数变量代替 |

| 关键字 | 区域 | 大小 | 适用场景 |
