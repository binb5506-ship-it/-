```mermaid
flowchart TB
    subgraph 初始化
        init["系统初始化\n1.关闭蜂鸣器等外设\n2.数码管LED熄灭"]
    end

    subgraph 主控制循环
        run["运行状态"] -->|"S7启动/停止"| stop["停止状态"]
        stop -->|"S7启动/停止"| run
        run --> modes{{"工作模式"}}
        
        modes -->|"模式1"| mode1["从左到右\nL1->L2->...->L8"]
        modes -->|"模式2"| mode2["从右到左\nL8->L7->...->L1"]
        modes -->|"模式3"| mode3["L1,L8->L2,L7->L3,L6->L4,L5"]
        modes -->|"模式4"| mode4["L4,L5->L3,L6->L2,L7->L1,L8"]
    end

    subgraph 参数设置
        run -->|"S6设置"| config["设置界面\n0.8s闪烁"]
        config -->|"S6循环"| modeSet["运行模式设置"]
        modeSet -->|"S6循环"| timeSet["流转间隔设置\n400ms-1200ms"]
        timeSet -->|"S6循环"| run

        modeSet -->|"S5/S4"| modePlus["模式+1/-1"]
        timeSet -->|"S5/S4"| timePlus["间隔±100ms"]
    end

    subgraph 亮度控制
        brightness["亮度等级(1-4)\nRB2: 0V-5V"] 
        run -->|"S4按下"| showLevel["显示亮度等级"]
        showLevel -->|"S4松开"| run
    end

    subgraph 数据存储
        E2PROM["E2PROM存储\n1.工作模式\n2.流转间隔"]
    end

    style run fill:#f96,stroke:#333,stroke-width:4px
    style config fill:#9cf,stroke:#333,stroke-width:2px
    style modes fill:#ff9,stroke:#333,stroke-width:2px
    style E2PROM fill:#f9f,stroke:#333,stroke-width:2px
```
